\chapter*{ List of Symbols}
\addcontentsline{toc}{chapter}{Symbols}

\noindent
\begin{longtable}{@{}l @{\quad=\quad} l@{}}
	$\theta$    & Compression ramp angle  \\
	$\phi$    &  Angle of sweep \\
	$C_f$  & Coefficient of skin friction \\
	$L_s$   &Length of separation  \\
	$x,y,z$  & Stream-wise, Wall normal and span-wise direction \\
	$\Delta t$   & Time step \\
	$\Omega$          & Volume of the cell \\
	$\Omega S$              & Surface area of a face of the cell \\
	$\vec{F}_c$       & Convective flux vector \\
	$\vec{F}_v$       & Viscous flux vector \\
	$\rho$            & Density \\
	$E$               & Energy \\
	$H$               & Enthalpy \\
	$u_i$             & Components of velocity \\
	$n_i$             & Components of unit normal \\
	$V$               & Contravariant velocity \\
	$\tau_{ij}$       & Shear stress \\
	$\mu$             & Dynamic viscosity \\
	$Re$              & Reynolds number \\
	$\delta_{ij}$     & Kronecker delta function \\
	$\Theta$          & Work done by viscous stress and heat conduction \\
	$T$               & Temperature \\
	$k$               & Thermal conductivity of the fluid (W/m·K) \\
	$\gamma$          & Ratio of specific heats (dimensionless) \\
	$Pr$              & Prandtl number (dimensionless, 0.71 in this work) \\
	$R$               & Non-dimensionalized gas constant \\
	$M_\infty$        & Free stream Mach number (dimensionless) \\
	$S$               & Sutherland’s constant \\

	
\end{longtable}
